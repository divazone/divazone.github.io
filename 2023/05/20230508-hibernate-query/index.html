@Override

public SysUser findUserByLoginName(String pLoginName) {
    String hql = "from SysUser as u where u.loginName = ?";
    List<SysUser> users = getHibernateTemplate().find(hql, pLoginName);
    return users.isEmpty() ? null : users.get(0);
}
